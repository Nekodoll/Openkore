automacro AutoBuff {
	StatusInactiveHandle EFST_BLESSING
	JobIDNot 0
	exclusive 1
	timeout 20
	call {
	#do respawn
	switch ($.map) {
	case (== morocc) {
	do move morocc &rand(150,160) &rand(100, 105)
	do talknpc 158 107 n
	}
	case (== payon) {
	do move payon &rand(95,105) &rand(110, 120)
	do talknpc 100 116 n
	}
	}

}	
}
