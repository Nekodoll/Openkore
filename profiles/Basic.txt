automacro DealPay_fild07 {
	CharCurrentWeight > 45%
	InMap pay_fild07,mjolnir_10,morocc
	JobID 7
	exclusive 1
	timeout 400
	call {
		do ai manual
		do respawn
		do move morocc &rand(154,155) &rand(102, 104)
		pause 2
	@tmpINV = (usable,card,etc,etc,equip,usable)
	$j = 0
	while ($j < 6) {
	@tmpI = &InventoryType ($tmpINV[$j])
	if ($tmpI[0] > -1) {
		do deal Mirei:
		pause 2
		$i = 0
		while ($i < @tmpI && $i<10) {
    		do deal add $tmpI[$i]
    		$i++
    		}
		do deal
		pause 2
		do deal

	}
	$j++
	}	
	do ai on
}
}

	SimpleHookEvent AI/lockMap
	InMap morocc
	overrideAI 1
	exclusive 1
	timeout 5
	call {
		do ai manual
		do move morocc 154 103
		do talknpc 158 107 n
		do conf lockMap none 
		pause 2
		do ai on
		switch (&invamount (Butterfly wing)) {
		case (< 2) {
			do pm Mirei: Bwing
		}
		case (> 2) {
		pause 3
			call ttc
		}
		}
		

}
}
